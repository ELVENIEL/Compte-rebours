<body>
    <div id="compteRebours">00:15:00</div>
    <button onclick="startCountdown()">Démarrer</button>
    <button onclick="stopCountdown()">Arrêter</button>
    <button onclick="resetCountdown()">Réinitialiser</button>

    <script>
        let countdownTimer;
        let countdownTime = 900; // Temps en secondes (ici 5 minutes)

        function startCountdown() {
            if (!countdownTimer) {
                countdownTimer = setInterval(updateCountdown, 1000);
            }
        }

        function stopCountdown() {
            clearInterval(countdownTimer);
            countdownTimer = null;
        }

        function resetCountdown() {
            stopCountdown();
            countdownTime = 900;
            document.getElementById('compteRebours').textContent = '00:05:00';
        }

        function updateCountdown() {
            if (countdownTime > 0) {
                countdownTime--;
                let minutes = Math.floor(countdownTime / 60);
                let seconds = countdownTime % 60;
                document.getElementById('compteRebours').textContent =
                    '00:' + (minutes < 10 ? '0' : '') + minutes + ':' + (seconds < 10 ? '0' : '') + seconds;
            } else {
                stopCountdown();
                alert('Le temps est écoulé!');
            }
        }
    </script>
